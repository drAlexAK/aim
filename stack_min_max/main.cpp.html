8    /// stack supports fast getting of minimum and maximum values
9    /// стек с быстрым получением максимального и минимального элемента
10   
11   template<class T>
12   struct stack_min_max{
13   private:
14       stack<T> st; /// primary stack
15       stack<T> st_min; /// minimum of the stack
16       stack<T> st_max; /// maximum of the stack
17   public:
18       void pop(){
19           st.pop();
20           st_min.pop();
21           st_max.pop();
22       }
23       T get_min(){
24           return st_min.top();
25       }
26       T get_max(){
27           return  st_max.top();
28       }
29       T top()
30       {
31           return st.top();
32       }
33       T push(T x){
34           st.push(x);
35           if(st_min.empty()) st_min.push(x);
36           if(st_max.empty()) st_max.push(x);
37   
38           st_min.push(min(x, st_min.top()));
39           st_max.push(max(x, st_max.top()));
40       }
41   };
42   
43   int main() {
44   
45       stack_min_max<int> a;
46   
47       string s;
48   
49       cout << "use push, pop, get or exit commands\n";
50   
51       do {
52           int tmp;
53           cin >> s;
54           if(s == "pop") {
55               cout << a.top() << "\n";
56               a.pop();
57           }
58           else if(s == "push"){
59               cin >> tmp;
60               a.push(tmp);
61           }else if (s == "get"){
62               cout << "min: " << a.get_min() << "\n";
63               cout << "max: " << a.get_max() << "\n";
64           }else{break;}
65       } while(true);
66   
67       return 0;
